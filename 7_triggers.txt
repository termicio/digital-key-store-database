USE DIGITAL_KEY_STORE;

DELIMITER $$

CREATE TRIGGER TR_GROUP_MEMBER_ONE_OWNER_INS
BEFORE INSERT ON GROUP_MEMBER
FOR EACH ROW
BEGIN
  IF NEW.role = 'OWNER' THEN
    IF (SELECT COUNT(*) FROM GROUP_MEMBER WHERE group_id = NEW.group_id AND role = 'OWNER') > 0 THEN
      SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'A group can have only one OWNER.';
    END IF;
  END IF;
END$$

CREATE TRIGGER TR_GROUP_MEMBER_ONE_OWNER_UPD
BEFORE UPDATE ON GROUP_MEMBER
FOR EACH ROW
BEGIN
  IF NEW.role = 'OWNER' AND OLD.role <> 'OWNER' THEN
    IF (SELECT COUNT(*) FROM GROUP_MEMBER WHERE group_id = NEW.group_id AND role = 'OWNER') > 0 THEN
      SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'A group can have only one OWNER.';
    END IF;
  END IF;
END$$

CREATE TRIGGER TR_GROUP_MEMBER_NO_EMPTY_GROUP_DEL
BEFORE DELETE ON GROUP_MEMBER
FOR EACH ROW
BEGIN
  IF COALESCE(@ALLOW_GROUP_DELETE, 0) = 0 THEN
    IF (SELECT COUNT(*) FROM GROUP_MEMBER WHERE group_id = OLD.group_id) = 1 THEN
      SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cannot delete the last member of a group (group cannot be empty).';
    END IF;
  END IF;
END$$

CREATE TRIGGER TR_CART_ITEM_CART_ACTIVE
BEFORE INSERT ON CART_ITEM
FOR EACH ROW
BEGIN
  DECLARE v_status VARCHAR(20);
  SELECT status INTO v_status FROM CART WHERE cart_id = NEW.cart_id;
  IF v_status <> 'ACTIVE' THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cannot add items to an inactive cart.';
  END IF;
END$$

CREATE TRIGGER TR_ACCESS_REQUIRES_PURCHASE
BEFORE INSERT ON GROUP_KEY_ACCESS
FOR EACH ROW
BEGIN
  IF (SELECT COUNT(*) FROM ENTITLEMENT WHERE code = NEW.code AND status = 'ACTIVE') = 0 THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Cannot share a key that has not been purchased.';
  END IF;
END$$

DELIMITER ;
