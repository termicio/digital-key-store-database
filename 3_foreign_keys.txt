USE DIGITAL_KEY_STORE;

ALTER TABLE LICENSE_KEY
  ADD CONSTRAINT LICENSE_KEY_FK1
  FOREIGN KEY (sku) REFERENCES PRODUCT (sku)
  ON UPDATE CASCADE
  ON DELETE RESTRICT;

ALTER TABLE CART
  ADD CONSTRAINT CART_FK1
  FOREIGN KEY (email) REFERENCES USER_ACCOUNT (email)
  ON UPDATE CASCADE
  ON DELETE RESTRICT;

ALTER TABLE CART_ITEM
  ADD CONSTRAINT CART_ITEM_FK1
  FOREIGN KEY (cart_id) REFERENCES CART (cart_id)
  ON UPDATE CASCADE
  ON DELETE CASCADE,
  ADD CONSTRAINT CART_ITEM_FK2
  FOREIGN KEY (sku) REFERENCES PRODUCT (sku)
  ON UPDATE CASCADE
  ON DELETE RESTRICT;

ALTER TABLE ORDERS
  ADD CONSTRAINT ORDERS_FK1
  FOREIGN KEY (email) REFERENCES USER_ACCOUNT (email)
  ON UPDATE CASCADE
  ON DELETE RESTRICT;

ALTER TABLE ENTITLEMENT
  ADD CONSTRAINT ENTITLEMENT_FK1
  FOREIGN KEY (order_number) REFERENCES ORDERS (order_number)
  ON UPDATE CASCADE
  ON DELETE CASCADE,
  ADD CONSTRAINT ENTITLEMENT_FK2
  FOREIGN KEY (code) REFERENCES LICENSE_KEY (code)
  ON UPDATE CASCADE
  ON DELETE RESTRICT;

ALTER TABLE GROUP_MEMBER

  FOREIGN KEY (group_id) REFERENCES SHARING_GROUP (group_id)
  ON UPDATE CASCADE
  ON DELETE RESTRICT,
  ADD CONSTRAINT GROUP_MEMBER_FK2
  FOREIGN KEY (email) REFERENCES USER_ACCOUNT (email)
  ON UPDATE CASCADE
  ON DELETE RESTRICT;

ALTER TABLE GROUP_KEY_ACCESS
  ADD CONSTRAINT GROUP_KEY_ACCESS_FK1
  FOREIGN KEY (group_id) REFERENCES SHARING_GROUP (group_id)
  ON UPDATE CASCADE
